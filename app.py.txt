import streamlit as st
import pandas as pd
import networkx as nx

# --- CONFIGURACIÓN DE PÁGINA ---
st.set_page_config(page_title="LEGO Swap Asociación", page_icon="🧩")

# Enlaces de tu Google Sheets (Ya configurados)
URL_INV = "https://docs.google.com/spreadsheets/d/e/2PACX-1vRgUeWKSvV8S20NodEnV3RMVQtE3xQk84NgDEnSpBd9knQY8MxyrcOgCPO9lQSHlmrPjLecm5NuUiAA/pub?gid=1446180612&single=true&output=csv"
URL_DES = "https://docs.google.com/spreadsheets/d/e/2PACX-1vRgUeWKSvV8S20NodEnV3RMVQtE3xQk84NgDEnSpBd9knQY8MxyrcOgCPO9lQSHlmrPjLecm5NuUiAA/pub?gid=119622631&single=true&output=csv"

@st.cache_data(ttl=60) # Actualiza datos cada minuto
def cargar_datos():
    inv = pd.read_csv(URL_INV)
    des = pd.read_csv(URL_DES)
    inv.columns = inv.columns.str.strip()
    des.columns = des.columns.str.strip()
    return inv, des

st.title("🧩 LEGO Swap: Intercambio de Sets")
st.markdown("Bienvenido a la plataforma de gestión de intercambios de la asociación.")

inv, des = cargar_datos()

tab1, tab2, tab3 = st.tabs(["📦 El Muro (Sets)", "❤️ Mis Deseos", "🚀 Resultados"])

with tab1:
    st.header("Sets Disponibles para Cambio")
    st.dataframe(inv, use_container_width=True)
    st.info("💡 Anota el SetID de los que te gusten para marcarlos en la siguiente pestaña.")

with tab2:
    st.header("Registrar mis Preferencias")
    st.write("Actualmente, este prototipo lee tus deseos directamente del Excel.")
    st.warning("Próxima mejora: Añadir botón para marcar deseos desde aquí mismo.")
    st.write("Tus deseos actuales registrados:")
    st.table(des)

with tab3:
    st.header("Algoritmo de Maximización")
    if st.button("🔄 Calcular Intercambios Óptimos"):
        G = nx.DiGraph()
        for _, row in des.iterrows():
            pide, set_id = str(row['Socio']), str(row['SetID'])
            duenos = inv[inv['SetID'].astype(str) == set_id]
            for _, d in duenos.iterrows():
                if pide != str(d['Socio']):
                    G.add_edge(str(d['Socio']), pide, set_id=set_id)

        ciclos = sorted(list(nx.simple_cycles(G)), key=len, reverse=True)
        
        if not ciclos:
            st.error("No se han encontrado ciclos de intercambio todavía.")
        else:
            usados = set()
            for i, ciclo in enumerate(ciclos, 1):
                if any(s in usados for s in ciclo): continue
                
                with st.expander(f"✅ Propuesta #{i}: Intercambio de {len(ciclo)} personas", expanded=True):
                    for j in range(len(ciclo)):
                        dante, recp = ciclo[j], ciclo[(j+1)%len(ciclo)]
                        st.write(f"👤 **{dante}** entrega Set **{G[dante][recp]['set_id']}** ➡️ a **{recp}**")
                for s in ciclo: usados.add(s)
            st.success(f"¡Total de {len(usados)} socios felices!")
